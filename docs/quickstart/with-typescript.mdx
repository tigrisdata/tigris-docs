# Quickstart: TypeScript

The following guide will get you up and running with TypeScript.

:::tip
Define the data model in the application code, which sets up the database
instantaneously. Easily evolve the data model as your application evolves
without worrying about breaking a rigid schema.

Quickly add data and easily retrieve or edit that data through simple and
intuitive APIs. Continue using your preferred programming language, there is
no new database query language to learn.

:::

First let's review what you need to get going.

## Prerequisites

All you'll need is any computer running macOS or Linux, with the following
installed:

- [Docker](https://docs.docker.com/get-docker/)
- [NodeJS](https://nodejs.org/en/) v12 or later

## Quickstart

```typescript title=app.ts
import { Collection, DB, Tigris } from "@tigrisdata/core";
import {
  TigrisCollectionType,
  TigrisDataTypes,
  TigrisSchema,
} from "@tigrisdata/core/dist/types";

export class Application {
  private readonly tigris: Tigris;

  constructor() {
    // initialize the client
    this.tigris = new Tigris({
      serverUrl: "localhost:8081",
    });

    this.tigrisPlayground();
  }

  public async tigrisPlayground() {
    // create the database if it does not exist,
    // and fetch the database object
    const db: DB = await this.tigris.createDatabaseIfNotExists("hello-db");

    // create the collections if they don't exist, or
    // update the collection schema
    const users: Collection<User> = await db.createOrUpdateCollection(
      "users",
      userSchema
    );

    // insert
    const jania: User = await users.insert({
      name: "Jania McGrory",
      balance: 6045.7,
    });
    const bunny: User = await users.insert({
      name: "Bunny Instone",
      balance: 2948.87,
    });

    // find the user by primary key field
    const user1: User = await users.readOne({
      userId: jania.userId,
    });
    const user2: User = await users.readOne({
      userId: bunny.userId,
    });

    // update
    await users.update(
      {
        userId: bunny.userId,
      },
      {
        name: "Jania McGrover",
      }
    );

    // transaction - transfer balance between users
    await db.transact(async (tx) => {
      // update balance
      await users.update(
        {
          userId: jania.userId,
        },
        {
          balance: jania.balance - 100,
        },
        tx
      );

      await users.update(
        {
          userId: bunny.userId,
        },
        {
          balance: bunny.balance + 100,
        },
        tx
      );
    });

    // delete users
    await users.delete({
      userId: jania.userId,
    });
    await users.delete({
      userId: bunny.userId,
    });

    // drop database
    await this.tigris.dropDatabase("hello-db");
  }
}

// declare the data model
interface User extends TigrisCollectionType {
  userId?: number;
  name: string;
  balance: number;
}

const userSchema: TigrisSchema<User> = {
  userId: {
    type: TigrisDataTypes.INT32,
    primary_key: {
      order: 1,
      autoGenerate: true,
    },
  },
  name: {
    type: TigrisDataTypes.STRING,
  },
  balance: {
    type: TigrisDataTypes.NUMBER,
  },
};

new Application();
```
